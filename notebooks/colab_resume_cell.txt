# ── DIAGNOSE + PULL FROM DRIVE (paste this once at the top of the new VM) ──
import os, pathlib, subprocess, shutil

# 0. Mount Drive (no-op if already mounted)
from google.colab import drive
drive.mount('/content/drive', force_remount=False)

# 1. Locate backup
BACKUP = pathlib.Path('/content/drive/MyDrive/radical_llm_backup')
print('Backup folder exists:', BACKUP.exists())
if BACKUP.exists():
    for sub in ['embeddings', 'embeddings_iso']:
        p = BACKUP / sub
        if p.exists():
            n = sum(1 for _ in p.iterdir())
            sz = subprocess.check_output(['du','-sh', str(p)]).decode().split()[0]
            print(f'  {sub}: {n} model folders, {sz}')
        else:
            print(f'  {sub}: MISSING')
else:
    print('STOP: no backup at', BACKUP)
    print('If you used a different account for backup, share that folder')
    print('"Anyone with the link" then visit the link from THIS account and')
    print('right-click the folder -> "Add shortcut to Drive" -> MyDrive.')

# 2. Clone repo + install (only if not already done)
if not pathlib.Path('/content/repo').exists():
    subprocess.run(['git','clone','https://github.com/aryan35790jp/chinese_llm.git','/content/repo'], check=True)
os.chdir('/content/repo')
subprocess.run(['pip','install','-q','transformers==4.44.2','tokenizers==0.19.1',
    'numpy==1.26.4','pandas==2.2.2','scipy==1.13.1','scikit-learn==1.5.1',
    'matplotlib==3.9.0','tqdm==4.66.4','datasets==2.20.0','sentencepiece==0.2.0',
    'Pillow==10.4.0','umap-learn==0.5.6',
    'fugashi==1.3.2','unidic-lite==1.0.8','ipadic==1.0.0'])

# 3. Pull cache FROM Drive TO local
if BACKUP.exists():
    pathlib.Path('/content/repo/cache').mkdir(exist_ok=True)
    for sub in ['embeddings', 'embeddings_iso']:
        src = BACKUP / sub
        dst = pathlib.Path('/content/repo/cache') / sub
        if src.exists():
            print(f'Copying {sub} (this takes a few minutes)...')
            # rsync is fastest, falls back to cp if rsync flaky on Drive FUSE
            r = subprocess.run(['rsync','-a','--info=progress2',str(src)+'/',str(dst)+'/'])
            if r.returncode != 0:
                shutil.copytree(src, dst, dirs_exist_ok=True)
            print(f'  done -> {dst}')

# 4. Sanity check
iso = pathlib.Path('/content/repo/cache/embeddings_iso')
if iso.exists():
    print('\nModels with iso-corrected embeddings on this VM:')
    for d in sorted(iso.iterdir()):
        print(' ', d.name)
